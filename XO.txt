void reset(char a,char b,char M[3][3],bool select[3][3],char XO,bool inuse[3][3]){

		for(int j=0;j<3;j++){
			M[i][j]='#';
		}
	}
		for(int i=0;i<3;i++){
		for(int j=0;j<3;j++){
			select[i][j]=0;
		}
		for(int i=0;i<3;i++){
		for(int j=0;j<3;j++){
			inuse[i][j]=0;
		}}}

}
